# Every page on this site renders on demand and reads Firestore to do it,
# so an unthrottled crawler costs real money. In July 2026 crawler traffic
# was running ~25x human traffic and accounted for ~96% of a $34/mo
# Firestore read bill. Hence the rules below.
#
# The intent is: let search engines index the pages a person would search
# for (home, teams, standings, schedule, scores, archive) and keep them
# out of the deep per-game and per-player pages, which are numerous,
# expensive to render, and worthless as search results.

User-agent: *
Crawl-delay: 10

# Private / authenticated areas — no reason to ever crawl these.
Disallow: /api/
Disallow: /admin
Disallow: /captain
Disallow: /score/
Disallow: /_platform

# Deep, high-cardinality pages. Hundreds of URLs, each a full render.
Disallow: /games/
Disallow: /players/

# Query-string variants multiply the same page into many crawlable URLs.
Disallow: /*?tab=
Disallow: /*?season=

# Commercial SEO/AI scrapers: all cost, no benefit to a rec-league site.
User-agent: AhrefsBot
Disallow: /
User-agent: SemrushBot
Disallow: /
User-agent: MJ12bot
Disallow: /
User-agent: DotBot
Disallow: /
User-agent: DataForSeoBot
Disallow: /
User-agent: BLEXBot
Disallow: /
User-agent: PetalBot
Disallow: /
User-agent: Bytespider
Disallow: /
